class Solution {
private:
    int n,m,k;
    bool vis[1005][1005];
    int dx[4]={1,-1,0,0};
    int dy[4]={0,0,-1,1};
    int ans=0;
    int D(int x){
        int ret=0;
        while(x>0){
            ret+=x%10;
            x/=10;
        }
        return ret;
    }
    void dfs(int x,int y){
        if(vis[x][y]==true)
            return ;
        if(k>=0)
        ans++;
        vis[x][y]=true;
        for(int t=0;t<4;t++){
            int tx=x+dx[t];
            int ty=y+dy[t];
            int ok=0;
            ok+=D(tx);ok+=D(ty);
            if(ok<=k&&tx>=0&&ty>=0&&tx<n&&ty<m)
                dfs(tx,ty);
        }
    }
public:
    int movingCount(int threshold, int rows, int cols)
    {
        memset(this->vis,false,sizeof this->vis);
        this->n=rows;
        this->m=cols;
        this->k=threshold;
        dfs(0,0);
        return ans;
    }
};
