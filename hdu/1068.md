### 二分图（无向图）最大独立数：选取最多的点，使任意所选两点均不相连
#### 有向图求法：有向图的最大独立集数 生成二分图 有向图的最大独立集数=顶点数 - (二分图最大匹配数)/2
```
#include <bits/stdc++.h>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>

#include<map>
#include<sstream>
#define ll long long
#define LL long long
#define inf 1000000000
#define iosfalse ios::sync_with_stdio(false);
#define lowbit(x) x&-x
using namespace std;

struct Edge
{
    int from;
    int to;
    int weight;
    Edge(int f, int t, int w):from(f), to(t), weight(w) {}
};
const int __maxNodes=3000;
vector<int> G[__maxNodes]; //G[i] 存储顶点 i 出发的边的编号
vector<Edge> edges; ///注意while loop 时 要clear
typedef vector<int>::iterator iterator_t;
int num_nodes;
int num_left;
int num_right;
int num_edges;
queue<int> Q;
int prevs[__maxNodes];
int matching[__maxNodes];
int check[__maxNodes];

int Hungarian()  //稀疏图更快
{
    int ans = 0;
    memset(matching, -1, sizeof(matching));
    memset(check, -1, sizeof(check));
    memset(prevs,-1,sizeof(prevs));
    for (int i=0; i<num_left; ++i) {
        if (matching[i] == -1) {
            while (!Q.empty()) Q.pop();
            Q.push(i);
            prevs[i] = -1; // 设 i 为路径起点
            bool flag = false; // 尚未找到增广路
            while (!Q.empty() && !flag) {
                int u = Q.front();
                for (iterator_t ix = G[u].begin(); ix != G[u].end() && !flag; ++ix) {
                    int v = edges[*ix].to;
                    if (check[v] != i) {
                        check[v] = i;
                        Q.push(matching[v]);
                        if (matching[v] >= 0) { // 此点为匹配点
                            prevs[matching[v]] = u;
                        } else { // 找到未匹配点，交替路变为增广路
                            flag = true;
                            int d=u, e=v;
                            while (d != -1) {
                                int t = matching[d];
                                matching[d] = e;
                                matching[e] = d;
                                d = prevs[d];
                                e = t;
                            }
                        }
                    }
                }
                Q.pop();
            }
            if (matching[i] != -1) ++ans;
        }
    }
    return ans;
}
int main()
{
    while(cin>>num_left)
    {
        int idx=0;
        edges.clear();
        num_nodes=num_left*2;
        for(int i=0;i<num_nodes;i++)
        {
            G[i].clear();
        }
        for(int i=0;i<num_left; i++)
        {
            int num;
            scanf("%d: (%d)",&num,&num);
            for(int j=0;j<num;j++)
            {
            int u,v;
            u=i;
            cin>>v;
            edges.push_back(Edge(u,v+num_left,1));
            G[u].push_back(idx++);
            edges.push_back(Edge(v+num_left,u,1));
            G[v+num_nodes].push_back(idx++);
            }
        }
        cout<<(num_nodes-Hungarian())/2<<endl;
    }
    return 0;
}
```
