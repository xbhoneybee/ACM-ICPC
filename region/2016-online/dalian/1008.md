对于m个query 可以存下对于之前的l r的答案。当l相同时，若r2>r1,<l,r2>的答案可以从<l,r1>继承接在r1后面算。

  //水数据 我这O(N^2)也能过
  int n,m;
  int a[100005];
  int sub[100005];
  int main()
  {
      int t;
      scanf("%d",&t);
      while(t--)
      {
          memset(a,0,sizeof a);
          scanf("%d",&n);
          for(int i=1;i<=n;i++)
              {
                  sub[i]=inf;
                  scanf("%d",&a[i]);
              }
          for(int i=1;i<=n;i++)//找到其后第一个比其小的数
          {
              for(int j=i+1;j<=n;j++)
              {
                  if(a[i]>a[j])
                  {
                      sub[i]=j;
                      break;
                  }
              }
          }
          scanf("%d",&m);
          for(int i=0;i<m;i++)
          {
              int l,r;
              scanf("%d %d",&l,&r);
              int tmp=a[l];
              while(l<=r)
              {
                  if(sub[l]>r)
                  {
                    printf("%d\n",tmp);
                      break;
                  }
                  tmp=tmp%a[sub[l]];
                  l=sub[l];
              }
          }
      }
      //printf("Time used = %.8f\n",(double)clock()/CLOCKS_PER_SEC);
      return 0;
  }
